Sistema de Gestión de Biblioteca

Descripción:
Aplicacion de escritorio para la gestión de biblioteca: prestamo de libros, devoluciones y registro de usuarios.
Como lenguaje de programacion usaré Java y para hacer la aplicacion el programa InteliJ, tambien usaré Maven para importar las librerias.
Como base de datos usaré MySQL y para gestionarla quiero usar HeidiSQL.
Las tablas seran para gestionar libros, autores, usuarios, prestamos y devoluciones.

Características que se pueden añadir:
Registrar libros, usuarios, autores, préstamos y devoluciones.
Libros con diferentes estados para saber si estan disponibles para prestar o no.
Busqueda de libros por titulo o ISBN.

Busqueda de usuarios por DNI o email.
Alertas por devoluciones atrasadas y multas si el retraso es muy grande.
Historial de prestamos a usuarios y libros.
Generar reportes mensuales de prestamos y devoluciones.

================================================================================================================================================================

Tablas:
Usuarios: DNI, nombre, apellido, email, telefono, direccion, fecha_de_nacimiento, fecha_de_registro
Autores: ID, nombre, apellido, pais, fecha_de_nacimiento
Libros: ID, ISBN, titulo, autor, categoria, editorial, numero_paginas, idioma, anio_publicacion, estado(disponible, prestado, deteriorado, bloqueado)
Prestamos: ID, usuario, libro, fecha_prestamo, fecha_devolucion, fecha_devolucion_real, multa



CREATE DATABASE IF NOT EXISTS biblioteca;
USE biblioteca;

CREATE TABLE Usuarios (
    DNI VARCHAR(20) PRIMARY KEY,
    nombre VARCHAR(50) NOT NULL,
    apellido VARCHAR(50) NOT NULL,
    email VARCHAR(100) NOT NULL UNIQUE,
    telefono VARCHAR(15) NOT NULL,
    direccion VARCHAR(255) NOT NULL,
    fecha_de_nacimiento DATE NOT NULL,
    fecha_de_registro DATE DEFAULT CURRENT_DATE
);

CREATE TABLE Autores (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(50) NOT NULL,
    apellido VARCHAR(50) NOT NULL,
    pais VARCHAR(50) NOT NULL,
    fecha_de_nacimiento DATE NOT NULL
);

CREATE TABLE Libros (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    ISBN VARCHAR(20) UNIQUE NOT NULL,
    titulo VARCHAR(255) NOT NULL,
    autor INT NOT NULL,
    categoria VARCHAR(50) NOT NULL,
    editorial VARCHAR(100) NOT NULL,
    numero_paginas INT NOT NULL,
    idioma VARCHAR(50) NOT NULL,
    anio_publicacion YEAR NOT NULL,
    estado ENUM('disponible', 'prestado', 'deteriorado', 'bloqueado') DEFAULT 'disponible',
    FOREIGN KEY (autor) REFERENCES Autores(ID) ON DELETE CASCADE
);

CREATE TABLE Prestamos (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    usuario VARCHAR(20) NOT NULL,
    libro INT NOT NULL,
    fecha_prestamo DATE NOT NULL,
    fecha_devolucion DATE NOT NULL,
    fecha_devolucion_real DATE,
    multa DECIMAL(10, 2) DEFAULT 0.00,
    FOREIGN KEY (usuario) REFERENCES Usuarios(DNI) ON DELETE CASCADE,
    FOREIGN KEY (libro) REFERENCES Libros(ID) ON DELETE CASCADE
);
